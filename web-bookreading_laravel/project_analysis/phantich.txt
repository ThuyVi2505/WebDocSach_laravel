        - category_description -> text
        - category_image -> varchar(255)
        - category_status -> tinyInteger -> default('0')
        - created_by -> integer
        - created_at -> timestamps
        - updated_at -> timestamps

        - id -> integer

        - author_slug -> varchar(255)
        - author_gender -> tinyInteger -> default('0-nữ, 1-nam')
        - author_overview -> text
        - author_image -> varchar(255)
        - author_status -> tinyInteger -> default('0')
        - created_at -> timestamps
        - updated_at -> timestamps
